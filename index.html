algoritmo clásico 
// Counting Sort en C (enteros no negativos)
#include <stdio.h>
#include <stdlib.h>
#include <limits.h>

void countingSort(int arr[], int n) {
    if (n <= 0) return;
    
    // 1. Encontrar el valor máximo
    int max = arr[0];
    for (int i = 1; i < n; i++)
        if (arr[i] > max) max = arr[i];
    
    // 2. Crear arreglo de conteo
    int *count = (int*)calloc(max + 1, sizeof(int));
    
    // 3. Contar ocurrencias
    for (int i = 0; i < n; i++)
        count[arr[i]]++;
    
    // 4. Reconstruir arreglo ordenado
    int index = 0;
    for (int i = 0; i <= max; i++) {
        while (count[i]--) {
            arr[index++] = i;
        }
    }
    free(count);
}

// Ejemplo de uso interactivo
int main() {
    int n;
    printf("Cantidad de elementos: ");
    scanf("%d", &n);
    
    int *arr = (int*)malloc(n * sizeof(int));
    printf("Ingrese %d enteros no negativos:\n", n);
    for (int i = 0; i < n; i++)
        scanf("%d", &arr[i]);
    
    countingSort(arr, n);
    
    printf("Arreglo ordenado: ");
    for (int i = 0; i < n; i++)
        printf("%d ", arr[i]);
    printf("\n");
    
    free(arr);
    return 0;
}
