Pseudocódigo

countingSort(A, k):
  // k = valor máximo en A
  count = new Array(k+1).fill(0)           ← Fase 1

  for i = 0 to A.length-1:                ← Fase 2
    count[A[i]]++

  for i = 1 to k:                          ← Fase 3
    count[i] += count[i-1]

  output = new Array(A.length)              ← Fase 4
  for i = A.length-1 downto 0:
    output[count[A[i]]-1] = A[i]
    count[A[i]]--

  return output
